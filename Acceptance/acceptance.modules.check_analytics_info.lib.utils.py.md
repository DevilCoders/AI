    # Defines a function named `collect_dump_data` that collects and processes data using a checker.

    # - `checker`: An instance of a checker class (e.g., `MetaChecker`, `TunnellerChecker`).
    # - `rows`: A list of rows, where each row is a dictionary representing data to be processed.

    row_dumps = [checker.process_row(row) for row in rows]
    # Uses a list comprehension to process each row in `rows` using the `process_row` method of the `checker`.
    # Stores the results (tuples of `verdict` and `message`) in the `row_dumps` list.

    result = checker.dump_result()
    # Calls the `dump_result` method of the `checker` to generate a summary of the checking process.
    # Stores the result (a dictionary containing the verdict and message) in the `result` variable.

    return {
        'result': result,
        # Returns the summary result generated by `dump_result`.
        'row_dumps': row_dumps
        # Returns the list of processed row results.
    }
